import random
from matplotlib import pyplot as plt

RandNums = set()

while len(RandNums) < 5:
    RandNums.add(random.randint(0, 75))
print("The Random Values Generated: ",RandNums)
XList = ['A', 'B', 'C', 'D', 'E']
RNList = list(RandNums)
PNList = [33, 2, 38, 47, 61]

plt.subplot(132)
plt.xlabel('Index Values')
plt.ylabel('Random Numbers')
plt.scatter(XList, RNList)
plt.subplot(133)
plt.xlabel('Index Values')

plt.suptitle('Random Number Plotting')
